/**
* Returns the normal deviate Z corresponding to a given lower tail area of `p`.
*
* @private
* @param {Probability} p - lower tail area
* @returns {number} normal deviate
*/
function apnorminv( p ) {

			(((((((b7 * r + b6) * r + b5) * r + b4) * r + b3) * r + b2) * r + b1) * r + 1.0);
	} else {
		if ( q < 0.0 ) {
